tree("Root")
    fallback
        tree("KitchenFire")
        tree("HandleNewCustomer")
        tree("TakeOrders")
        tree("ServeFood")
        tree("HandleComplaints")
        tree("RefillDrinks")
        tree("CleanTables")

tree("KitchenFire")
    sequence
        IsKitchenFireDetected
        MoveToKitchen
        ExtinguishFire
        ConfirmFireExtinguished

tree("HandleNewCustomer")
    sequence
        MoveToWaitingArea
        AskForTableRequirement
        AssignTable
        GuideToTable
        ProvideMenu

tree("TakeOrders")
    sequence
        IsServiceButtonPressed
        MoveToTable
        AskForOrder
        TakeOrder
        ConfirmOrder
        ProvideWaitTime

tree("ServeFood")
    sequence
        IsOrderReady
        MoveToKitchen
        PickUpFood
        MoveToTable
        ServeFood

tree("HandleComplaints")
    fallback
        sequence
            IsComplaintReceived
            Apologize
            RemakeFood
        sequence
            IsComplaintReceived
            CustomerLeaves

tree("RefillDrinks")
    sequence
        IsRefillRequestReceived
        MoveToTable
        PickUpDrink
        RefillDrink
        ReturnDrink

tree("CleanTables")
    sequence
        IsTableEmpty
        MoveToTable
        ClearTable
        CleanTable
